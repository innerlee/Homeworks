\subsection{} %4.1

Let $u = x+y, v=x-y$, then

\begin{align}
    \int_0^1\int_0^{1-x}\sqrt{x+y}(y-2x)^2dydx
        &= \int_0^1\int_{-u}^{u}\sqrt{u}\bigg(\frac{u-v}{2}-u-v\bigg)^2 \frac{1}{2} dvdu \\
        &= \frac{1}{8} \int_0^1\int_{-u}^{u} \sqrt{u} (u+3v)^2 dvdu \\
        &= \int_0^1 \sqrt{u} u^3 du \\
        &= \frac{2}{9}
\end{align}
